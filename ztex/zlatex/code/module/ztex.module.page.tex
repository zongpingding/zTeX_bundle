% ==> title page
\let\ori@maketitle\maketitle
\bool_if:NTF \g__ztex_slide_bool 
  {
    \newcommand\ztex@maketitle 
      {
        \bool_if:NT \g__ztex_hyperref_bool 
          {
            \phantomsection
            \hypertarget{zslide:titlepage}{}
          }
        \newgeometry{margin=1cm}
        \null\vfill\begin{center}
          \begin{tabular}{c}
            \begin{zpic}[unit=\textwidth]
              \zrectangle[arc=.01, draw=white, fill=zslide@title@color](-0.48, -.05)(.48, .05)
              \put(-.425, -.018){\hb@xt@.85\textwidth{\hss\Large\zslidetitle\hss}}
            \end{zpic}\\[3.5em]
            \zslideauthor\\[3em]
            \zslidedate
          \end{tabular}
        \end{center}\vfill\null
        \thispagestyle{empty}\setcounter{page}{0}
        \restoregeometry
      }
  }{
    \cs_generate_variant:Nn \ztool_get_ht:Nn {No}
    \long\def\format@title{{\huge\bfseries\@title}}
    \long\def\format@author{{\Large\bfseries\@author}}
    \long\def\format@date{{\Large\textcolor{gray}{\@date}}}
    \newcommand\title@upper@box[2][0pt]
      {
        \parbox[b][#2][r]{\l_tmpa_dim}{
          {\format@title}\\[#1]
          {\format@author}
        }
      }
    \newcommand\ztex@maketitle 
      {
        \thispagestyle{empty}
        % calc max width/height, add '1pt' for right padding in case of wrong line break
        \ztool_get_wd:Nn \l_tmpa_dim {\hbox:n {\format@title}}
        \ztool_get_wd:Nn \l_tmpb_dim {\hbox:n {\format@author}}
        \dim_set:Nn \l_tmpa_dim {
          \dim_min:nn {
            \dim_max:nn {\l_tmpa_dim}{\l_tmpb_dim} 
        }{.8\textwidth} + 1pt} 											              % the max title width
        \ztool_get_ht_plus_dp:Nn \l_tmpb_dim {\title@upper@box{}}
        \dim_set:Nn \l_tmpb_dim {\dim_max:nn {80pt}{\l_tmpb_dim}} % the total title height
        % typeset info
        \vfill\vspace*{20pt}\begin{center}
          \rule{6pt}{\l_tmpb_dim}\enskip
          \title@upper@box[\fill]{\l_tmpb_dim}
          \par\vfill\format@date
        \end{center}\newpage
      }
  }
\RenewDocumentCommand{\maketitle}{so}
  {
    \IfBooleanTF{#1}{\ori@maketitle}
      {
        \IfNoValueTF{#2}
          { \ztex@maketitle }
          {
            \newgeometry{margin=#2}
            \ori@maketitle
            \restoregeometry
          }
      }
  }


% ==> fancyhdr setup
\cs_new_protected:Npn \zpage_set_style:nnn #1#2#3
  {
    \tl_if_empty:nTF { #2 }
      { \fancypagestyle{#1}{#3} }
      { \fancypagestyle{#1}[#2]{#3} }
  }
\NewDocumentCommand{\zpagestyleset}{O{}mm}
  { 
    \zpage_set_style:nnn {#2}{#1}{#3}
    % \fancypagestyleassign {#1}{#1@new} 
    % ERROR: '! TeX capacity exceeded,' 
  }
\bool_if:NF \g__ztex_slide_bool
  {
    \RequirePackage{fancyhdr}
    \zpage_set_style:nnn { fancy }{}
      {
        \fancyhf{}
        \dim_gset:Nn \headheight{15pt}
        \renewcommand{\headrule}{\hrule width\textwidth}
        \if@twoside
          \fancyhead[EL]{\ztexleftmark}
          \fancyhead[ER]{\thepage}
          \fancyhead[OL]{\thepage}
          \fancyhead[OR]{\ztexrightmark}
        \else
          \IfClassLoadedTF{book}{
            \fancyhead[L]{\thepage}
            \fancyhead[R]{\ztexrightmark}
          }{
            \fancyhead[L]{\thepage}
            \bool_if:NTF \g__ztex_sect_load_bool
              {
                \fancyhead[R]
                  {
                    \IfMarksEqualTF{ztex-left}{first}{last}
                      { \FirstMark{ztex-left} }
                      { \FirstMark{ztex-left} \;\(\Rightarrow\)\; \LastMark{ztex-left} }
                  }
              }{
                \fancyhead[R]
                  {
                    \IfMarksEqualTF{2e-left}{first}{last}
                      { \FirstMark{2e-left} }
                      { \FirstMark{2e-left} \;\(\Rightarrow\)\; \LastMark{2e-left} }
                  }
              }
          }
        \fi
      }
    \zpage_set_style:nnn { plain }{}
      {
        \fancyhf{}
        \renewcommand{\headrulewidth}{0pt}
        \renewcommand{\headrule}{}
        \fancyfoot[C]{\thepage}
      }
  }

% ==> page annotation
% #1: fore/background; #2: position; 
% #3: anchor;          #4: object;
% #5: hook range.
\dim_const:Nn \zph {\paperheight}
\dim_const:Nn \zpw {\paperwidth}
\cs_generate_variant:Nn \hook_gput_code:nnn {nne}
\cs_new_protected:Npn \ztex_page_annotate:nnnnn #1#2#3#4#5
  {
    \tl_if_empty:eTF {#5}
      {
        \hook_gput_code:nnn {shipout/#1}
          {ztex@page@mask-\l__ztex_page_mask_label_tl}
          {\put#2{\makebox(0, 0)[#3]{#4}}}
      }{
        \hook_gput_next_code:nn {shipout/#1}
          {\put#2{\makebox(0, 0)[#3]{#4}}}
      }
  }
\hook_gset_rule:nnnn {shipout/foreground}
  { . }{ < }{ pgfrcs }
\hook_gset_rule:nnnn {shipout/background}
  { . }{ < }{ pgfrcs }
\ztex_keys_define:nn { page/mask }
  {
    layer    .choice:,
    layer / foreground  .code:n = 
      { \tl_set:Nn \l__ztex_page_mask_layer_tl {#1} },
    layer / background  .code:n = 
      { \tl_set:Nn \l__ztex_page_mask_layer_tl {#1} },
    layer / unknown     .code:n =
      {
        \ztex_msg_set:nn {pagemask@layer@wrong}
          {layer~only~accepts~values:'foreground'~or~'background'.}
        \ztex_msg_error:n {pagemask@layer@wrong}
      },
    layer    .initial:n = background,
    position .tl_set:N  = \l__ztex_page_mask_position_tl,
    position .initial:n = {(.5\zpw, .5\zph)},
    anchor   .tl_set:N  = \l__ztex_page_mask_anchor_tl,
    anchor   .initial:n = c,
    label    .tl_set:N  = \l__ztex_page_mask_label_tl,
    label    .initial:n = { DEFAULT },
  }
\cs_generate_variant:Nn \ztex_page_annotate:nnnnn {eee}
\cs_new:Npn \__page_mask_pos_parse:w (#1, #2) 
  {(
    \dim_to_decimal:n {#1} pt, 
    \dim_to_decimal:n {#2-\paperheight} pt
  )}
\ztex_msg_set:nn {pageinfo}
  {
    Only~star~version~of~\string\zpagemask\ 
    is~label-allowed.
  }
\prop_gclear_new:N \g__zpage_mask_label_prop
\NewDocumentCommand{\zpagemask}{so+m}
  {
    \group_begin:
    \IfValueT{ #2 }
      {
        \ztex_keys_set:nn { page/mask }{#2}
      }
    \IfBooleanTF{#1}{\gdef\@once@hook@sign{}}
      {
        \gdef\@once@hook@sign{*}
        \tl_if_eq:enF { \l__ztex_page_mask_label_tl }
          { DEFAULT }
          { \ztex_msg_warn:n {pageinfo} }
      }
    \exp_args:Neee \hook_gset_rule:nnnn
      { shipout/\l__ztex_page_mask_layer_tl }
      { ztex@page@mask-\l__ztex_page_mask_label_tl }
      { < }{ pgfrcs }
    \exp_args:NNe \prop_gput_from_keyval:Nn \g__zpage_mask_label_prop
      { \l__ztex_page_mask_label_tl = \l__ztex_page_mask_layer_tl }
    \ztex_page_annotate:eeenn
      { \l__ztex_page_mask_layer_tl }
      { \exp_after:wN \__page_mask_pos_parse:w \l__ztex_page_mask_position_tl }
      { \l__ztex_page_mask_anchor_tl }
      { #3 }{ \@once@hook@sign }
    \group_end:
  }
\ztex_msg_set:nn {zpage@masklabel@nonexist}
  {
    Page~mask~of~label:~'#1'~does~NOT~exist,~
    i'll~give~up~this~operation.~Current~page~
    mask~label~listed~as~follows(without~outer~braces):
    \prop_map_function:NN \g__zpage_mask_label_prop
      \__prop_typeout:nn
  }
\NewDocumentCommand{\zpagemaskrm}{ m }
  {
    \prop_if_in:NnTF \g__zpage_mask_label_prop
      { #1 }
      {
        \exp_args:Ne \hook_gremove_code:nn 
          { shipout/\prop_item:Nn \g__zpage_mask_label_prop {#1} }
          { ztex@page@mask-#1 }
        \prop_gpop:NnN \g__zpage_mask_label_prop
          { #1 } \l_tmpa_tl
      }{
        \ztex_msg_warn:nn 
          { zpage@masklabel@nonexist }
          { #1 }
      }
  }
\cs_new:Npn \__prop_typeout:nn #1#2
  { \\{#1}~~~=>~~~{#2} }
\NewDocumentCommand{\zpagemaskrule}{mmm}
  {
    \prop_if_in:NnTF \g__zpage_mask_label_prop {#1}
      {
        \prop_if_in:NnTF \g__zpage_mask_label_prop {#3}
          {
            \exp_args:Neee \hook_gset_rule:nnnn
              { shipout/\prop_item:Nn \g__zpage_mask_label_prop {#1} }
              { ztex@page@mask-#1 }
              { #2 }
              { ztex@page@mask-#3 }
          }{
            \ztex_msg_warn:nn
              { zpage@masklabel@nonexist }
              { #2 }
          }
      }{
        \ztex_msg_warn:nn
          { zpage@masklabel@nonexist }
          { #1 }
      }
  }
