    \RequirePackage{fontspec}
    \newfontfamily{\Cinzel}{CinzelRegular.ttf}[
      BoldFont=CinzelBold.ttf,
      ItalicFont=SabonItalic.ttf
    ]
  }{\def\Cinzel{\relax}}
}
