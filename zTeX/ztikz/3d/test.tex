\fp_const:Nn \Cw_fp {10} % Canvas width in pixels
\fp_const:Nn \Ch_fp {10} % Canvas height in pixels
\fp_const:Nn \Vw_fp {20} % Viewport width in units
\fp_const:Nn \Vh_fp {20} % Viewport height in units
\fp_const:Nn \d_fp  {3}  % Distance from the viewport to the canvas
%%% Symbols:
% 1. The canvas has a width and a height, measured in pixels, which weâ€™ll call 'Cw' and 'Ch'.
% 2. The frame has dimensions 'Vw' and 'Vh', at a distance 'd'; (z=d)
%%% Transformation Formulas:
% compute the projection of P on the viewport 'z=d';
% px' = px*d/(pz); py' = py*d/(pz);
% Cx = px'*Cw/Vw; Cy = py'*Cv/Vh;
\cs_set:Npn \ViewportToCanvas:nn #1#2
  {% #1:x, #2:y
    \fp_eval:n {#1*\Cw_fp/\Vw_fp},
    \fp_eval:n {#2*\Ch_fp/\Vh_fp}
  }
\cs_set:Npn \ProjectVertex:n #1 
  {% #1:(x, y, z)
    % I: use perspective projection
    \ViewportToCanvas:nn 
      {\clist_item:ne {#1}{1}*\d_fp/\clist_item:ne {#1}{3}}
      {\clist_item:ne {#1}{2}*\d_fp/\clist_item:ne {#1}{3}}

    % II: use isometric projection
    % \ViewportToCanvas:nn 
    %   {sqrt(3)/2*\clist_item:ne {#1}{1}-sqrt(3)/2*\clist_item:ne {#1}{3}}
    %   {(\clist_item:ne {#1}{2}+\clist_item:ne {#1}{3})/2+\clist_item:ne {#1}{2}}
  }
\cs_set:Npn \DrawLine:nnn #1#2#3 
  {
    \draw[#3] (#1) -- (#2);
  }
\cs_generate_variant:Nn \clist_item:nn {ne}
\cs_generate_variant:Nn \DrawLine:nnn {een}
\cs_generate_variant:Nn \ProjectVertex:n {o, e, V}
\ExplSyntaxOff
